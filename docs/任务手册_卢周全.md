**第二步：创建 BM25 检索模块**

创建文件 `src/vuln_analysis/knowledge/bm25_retriever.py`：

```python
"""BM25 稀疏检索模块 — 对 CVE 描述做关键词检索。"""

from rank_bm25 import BM25Okapi
from typing import Optional


class BM25Retriever:
    """基于 BM25 的 CVE 文本检索器。"""

    def __init__(self):
        self._index: Optional[BM25Okapi] = None
        self._documents: list[dict] = []
        self._corpus: list[list[str]] = []

    def build_index(self, cve_entries: list[dict]) -> None:
        """
        构建 BM25 索引。

        Parameters
        ----------
        cve_entries:
            列表，每项包含 {"cve_id": "CVE-xxx", "description": "漏洞描述..."}
        """
        self._documents = cve_entries
        self._corpus = [
            entry.get("description", "").lower().split()
            for entry in cve_entries
        ]
        self._index = BM25Okapi(self._corpus)

    def search(self, query: str, top_k: int = 5) -> list[dict]:
        """
        检索与 query 最相关的 CVE。

        Returns
        -------
        list[dict]
            [{"cve_id": "CVE-xxx", "description": "...", "score": 0.85}, ...]
        """
        if self._index is None:
            return []

        tokenized_query = query.lower().split()
        scores = self._index.get_scores(tokenized_query)

        # 取 top_k
        top_indices = sorted(
            range(len(scores)),
            key=lambda i: scores[i],
            reverse=True
        )[:top_k]

        results = []
        for i in top_indices:
            if scores[i] > 0:
                results.append({
                    "cve_id": self._documents[i].get("cve_id", ""),
                    "description": self._documents[i].get("description", ""),
                    "score": round(float(scores[i]), 4),
                })
        return results
```

from vuln_analysis.knowledge.bm25_retriever import BM25Retriever

## 🟡 任务二：混合检索融合（截止 3.16）

### 目标
将 BM25 + KnowledgeGraph 的结果合并排序，对外提供统一的检索接口。

### 步骤

创建 `src/vuln_analysis/knowledge/hybrid_retriever.py`：

```python
"""混合检索: BM25 文本检索 + KnowledgeGraph 图检索。"""

from vuln_analysis.knowledge.bm25_retriever import BM25Retriever
from vuln_analysis.knowledge.knowledge_graph import KnowledgeGraph


class HybridRetriever:
    """融合 BM25 + Graph 的双路检索器。"""

    def __init__(self, bm25: BM25Retriever, kg: KnowledgeGraph):
        self.bm25 = bm25
        self.kg = kg

    def search(self, query: str, cve_id: str = "", top_k: int = 5) -> list[dict]:
        """
        双路检索 + 倒数排名融合 (RRF)。

        1. BM25: 基于 query 文本做关键词检索
        2. Graph: 基于 cve_id 查找同 CWE 的相关漏洞
        3. RRF 融合: score = Σ 1/(k + rank)
        """
        scores = {}  # cve_id -> rrf_score
        k = 60  # RRF 常数

        # BM25 检索
        bm25_results = self.bm25.search(query, top_k=top_k * 2)
        for rank, item in enumerate(bm25_results):
            cid = item["cve_id"]
            scores[cid] = scores.get(cid, 0) + 1.0 / (k + rank)

        # Graph 检索 (如果提供了 cve_id)
        if cve_id:
            chain = self.kg.query_attack_chain(cve_id)
            if chain.get("found"):
                for cwe in chain.get("cwes", []):
                    related = self.kg.get_related_cves(cwe)
                    for rank, rcve in enumerate(related):
                        scores[rcve] = scores.get(rcve, 0) + 1.0 / (k + rank)

        # 按融合分数排序
        sorted_results = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        return [
            {"cve_id": cid, "rrf_score": round(score, 6)}
            for cid, score in sorted_results[:top_k]
        ]
```
